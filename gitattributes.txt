# Set default behavior to automatically normalize line endings.
* text=auto

# Declare files that will always have LF line endings on checkout.
*.sh eol=lf
*.yml eol=lf
*.yaml eol=lf

# Declare files that will always have CRLF line endings on checkout.
*.bat eol=crlf

# Denote all text files as LF line endings.
*.cs text eol=lf
*.js text eol=lf
*.html text eol=lf
*.css text eol=lf
*.json text eol=lf
*.md text eol=lf
*.txt text eol=lf

# Binary files should be left untouched.
*.png binary
*.jpg binary
*.jpeg binary
*.gif binary
*.mp4 binary
*.mp3 binary
*.wav binary
*.ogg binary
*.fbx binary
*.unity binary
*.prefab binary
